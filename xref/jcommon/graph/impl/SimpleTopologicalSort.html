24  import java.util.*;
25  import java.util.concurrent.Callable;

29  /**
30   * Please see the following for a description of this algorithm:
31   *   <a href="http://www.cs.washington.edu/education/courses/cse373/02sp/lectures/cse373-21-TopoSort-4up.pdf">http://www.cs.washington.edu/education/courses/cse373/02sp/lectures/cse373-21-TopoSort-4up.pdf</a>
32   */
33  public final class SimpleTopologicalSort<TVertex extends IVertex> implements ITopologicalSortStrategy<TVertex> {
34    @Override
35    public List<TVertex> sort(IAdjacencyList<TVertex> adjacencyList) throws CyclicGraphException {
36      if (adjacencyList.isEmpty())
37        return new ArrayList<TVertex>(0);
38  
39      int ordered_index = 0;
40      List<TVertex> ordered = new ArrayList<TVertex>(adjacencyList.size());
41  
42      Queue<IAdjacencyListPair<TVertex>> queue = new LinkedList<IAdjacencyListPair<TVertex>>();
43      int[] in_degrees = adjacencyList.calculateInDegrees();
44  
45      //Find all vertices who have an in-degree of zero.
46      for(int i = 0; i < in_degrees.length; ++i) {
47        if (in_degrees[i] == 0)
48          queue.add(adjacencyList.pairAt(i));
49      }
50  
51      //If there are no vertices with an in-degree of zero (meaning they have no one pointing to them),
52      //then this is not a DAG (directed acyclic graph).
53      if (queue.isEmpty())
54        throw new CyclicGraphException("Cycle detected when topologically sorting the graph");
55  
56      IAdjacencyListPair<TVertex> p;
57      while((p = queue.poll()) != null) {
58        ordered.add(p.getVertex());
59  
60        for(TVertex d: p.getOutNeighbors()) {
61          int index = adjacencyList.indexOf(d);
62          //Enqueue any vertex whose in-degree will become zero
63          if (in_degrees[index] == 1)
64            queue.add(adjacencyList.pairAt(index));
65          --in_degrees[index];
66        }
67      }
68  
69      //If we haven't filled the array, then there's a cycle somewhere.
70      if (ordered.size() != adjacencyList.size())
71        throw new CyclicGraphException("Cycle detected when topologically sorting the graph");
72  
73      return ordered;
74    }

76    @Override
77    public ITopologicalSortAsyncResult sortAsync(final ExecutorService executorProcessors, final IAdjacencyList<TVertex> adjacencyList, final ITopologicalSortCallback<TVertex> callback, final ITopologicalSortErrorCallback<TVertex> errorCallback) {
78      final TopologicalSortAsyncResult asyncResult = new TopologicalSortAsyncResult(executorProcessors);
79      if (adjacencyList.isEmpty()) {
80        asyncResult.asyncComplete(true);
81        return asyncResult;
82      }
83  
84      final Queue<Callable<Object>> queue = new LinkedList<Callable<Object>>();
85      final int[] in_degrees = adjacencyList.calculateInDegrees();
86      final ArrayList<Callable<Object>> callables = new ArrayList<Callable<Object>>(in_degrees.length);
87      final Set<IAdjacencyListPair<TVertex>> remaining = new HashSet<IAdjacencyListPair<TVertex>>();
88      final AtomicInteger[] atomics = new AtomicInteger[in_degrees.length];
89      final ArrayList<Map<TVertex, Object>> inputs = new ArrayList<Map<TVertex, Object>>(in_degrees.length);
90      final AtomicInteger outstanding_submissions = new AtomicInteger(0);
91      final ITopologicalSortCoordinator coordinator = new TopologicalSortCoordinator(asyncResult);
92  
93      //Find all vertices who have an in-degree of zero.
94      //Also initialize latches to the size of the the in-degree + 1.
95      for(int i = 0; i < in_degrees.length; ++i) {
96        final IAdjacencyListPair<TVertex> pair = adjacencyList.pairAt(i);
97        final AtomicInteger atom = atomics[i] = new AtomicInteger(in_degrees[i] == 0 ? 1 : in_degrees[i]);
98        final HashMap<TVertex, Object> my_input = new HashMap<TVertex, Object>(in_degrees[i], 1.0f);
99        inputs.add(my_input);
100 
101       //Ensure that we add all items in the adjacency list to our list of remaining items.
102       //As we discover the proper order to visit them and then process them, we will drain
103       //the list of remaining vertices. If there is no cycle, then the list of remaining
104       //vertices will be empty when we're through.
105       remaining.add(pair);
106 
107       final Callable<Object> callable = new Callable<Object>() {
108         @Override
109         @SuppressWarnings("unchecked")
110         public Object call() throws Exception {
111           Throwable handled_exception = null;
112           boolean handle_all_done = false;
113           TVertex vertex = pair.getVertex();
114           ITopologicalSortInput<TVertex> input;
115 
116           try {
117 
118             if (atom.decrementAndGet() == 0) {
119               //Remove from the list of remaining vertices.
120               synchronized (remaining) {
121                 remaining.remove(pair);
122               }
123 
124               synchronized (my_input) {
125                 input = new TopologicalSortInput<TVertex>(my_input);
126               }
127 
128               //Call the callback to let him handle this vertex.
129               Object result = null;
130               try {
131                 result = callback.handle(vertex, input, coordinator);
132               } catch(Throwable t) {
133                 //We need to handle the exception later after we've done other work.
134                 //Save it off for later evaluation.
135                 handled_exception = t;
136               }
137 
138               //Ensure we haven't been asked to stop processing. If so,
139               //we don't want to schedule anything else. We need to let the existing
140               //submissions drain.
141               if (!asyncResult.isProcessingDiscontinued()) {
142                 //Submit a task for everyone who is dependent on me.
143                 //On the next round if all of their vertices have been evaluated,
144                 //the count will be at zero.
145                 for(TVertex dep : pair.getOutNeighbors()) {
146                   final int index = adjacencyList.indexOf(dep);
147                   final Callable<Object> dep_callable = callables.get(index);
148                   final Map<TVertex, Object> dep_inputs = inputs.get(index);
149 
150                   synchronized(dep_inputs) {
151                     dep_inputs.put(vertex, result);
152                   }
153 
154                   //Ensure we track the number of submissions. This will be used to
155                   //detect an effective deadlock -- we can't make progress b/c there's
156                   //a cycle preventing it.
157                   outstanding_submissions.incrementAndGet();
158                   executorProcessors.submit(dep_callable);
159                 }
160               }
161             }
162 
163             //Check if we cannot make any more progress. We know to do this check b/c
164             //on every submission we atomically increment a counter and then when that
165             //submission is processed we decrement the counter. If the counter reaches
166             //zero, then we know we've effectively drained the submissions. At that
167             //point, if there are any remaining then we know there's a cycle.
168 
169             int outstanding = outstanding_submissions.decrementAndGet();
170 
171             if (outstanding == 0) {
172               //Save a notification for later processing after we've handled errors.
173               handle_all_done = true;
174 
175               if (remaining.size() > 0) {
176                 throw new CyclicGraphException("Cycle detected when topologically sorting the graph");
177               }
178             }
179 
180             //Now deal with any exception that was thrown upon processing the vertex.
181             //Will bubble up to the callable-wide handler below.
182             if (handled_exception != null)
183               throw handled_exception;
184 
185           } catch(Throwable t) {
186             if (errorCallback != null) {
187               try {
188                 errorCallback.handleError(vertex, t, coordinator);
189               } catch(Throwable t2) {
190                 //Swallow any exceptions thrown by our error handler.
191               }
192             }
193           }
194 
195           if (handle_all_done) {
196             asyncResult.asyncComplete(remaining.size() == 0);
197           }
198 
199           return null;
200         }
201       };
202       callables.add(callable);
203 
204       if (in_degrees[i] == 0) {
205         //We do this to ensure that callables has been fully
206         //initialized prior to starting up any tasks.
207         queue.add(callable);
208       }
209     }
210 
211     //If there are no vertices with an in-degree of zero (meaning they have no one pointing to them),
212     //then this is not a DAG (directed acyclic graph).
213     if (queue.isEmpty()) {
214       if (errorCallback != null) {
215         errorCallback.handleError(null, new CyclicGraphException("Cycle detected when topologically sorting the graph"), coordinator);
216       }
217       asyncResult.asyncComplete(false);
218       return asyncResult;
219     }
220 
221     //Initialize number of outstanding executor submissions.
222     //When this reaches zero, we need to check if there's deadlock (no path forward), which
223     //is a sign of a cycle.
224     outstanding_submissions.set(queue.size());
225 
226     Callable<Object> callable;
227     while((callable = queue.poll()) != null)
228       executorProcessors.submit(callable);
229 
230     return asyncResult;
231   }
