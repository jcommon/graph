20  package jcommon.graph.impl;
21  
22  import jcommon.graph.IEdge;
23  import jcommon.graph.IVertex;
24  
25  /**
26   * @param <TVertex> Type of {@link IVertex} instances that this edge will connect.
27   *
28   * @see IEdge
29   */
30  public class Edge<TVertex extends IVertex> implements IEdge<TVertex> {
31    /** The object we are connecting from. */
32    public final TVertex from;
33  
34    /** The object we are connecting to. */
35    public final TVertex to;
36  
37    /**
38     * Create a new directed edge for a graph that defines the relationship between
39     * two graph vertices.
40     *
41     * @param from The object we are connecting from.
42     * @param to The object we are connecting to.
43     */
44    public Edge(final TVertex from, final TVertex to) {
45      if (from == null || to == null)
46        throw new NullPointerException("from and to both cannot be null");
47  
48      this.from = from;
49      this.to = to;
50    }
51  
52    /**
53     * Retrieves the {@link Edge#from} instance.
54     *
55     * @return The provided {@link Edge#from} instance.
56     */
57    @Override
58    public TVertex getFrom() {
59      return from;
60    }
61  
62    /**
63     * Retrieves the {@link Edge#to} instance.
64     *
65     * @return The provided {@link Edge#to} instance.
66     */
67    @Override
68    public TVertex getTo() {
69      return to;
70    }
71  
72    /**
73     * @see Object#hashCode()
74     */
75    @Override
76    public int hashCode() {
77      int result = from.hashCode();
78      result = 31 * result + to.hashCode();
79      return result;
80    }
81  
82    /**
83     * @see Object#equals(Object)
84     */
85    @Override
86    public boolean equals(final Object obj) {
87      if (!(obj instanceof Edge))
88        return false;
89      final Edge e = (Edge)obj;
90  
91      return from.equals(e.from) && to.equals(e.to);
92    }
93  }
