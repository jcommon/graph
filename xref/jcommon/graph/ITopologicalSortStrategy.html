20  package jcommon.graph;
21  
22  import java.util.List;
23  import java.util.concurrent.ExecutorService;
24  
25  /**
26   * Provides a strategy for implementing a topological sort of an {@link IGraph}.
27   *
28   * A {@link IGraph} is first transformed into an adjacency list of type {@link IAdjacencyList}. That is then used
29   * to topologically sort the graph.
30   *
31   * Two versions are provided: one will synchronously traverse the graph and provide the ordered list of vertices to
32   * visit and the second will asynchronously process the graph executing in parallel as many as possible given the size
33   * of a thread pool and the ordering.
34   *
35   * @param <TVertex> Type of {@link IVertex} that a topological sort will operate on.
36   *
37   * @see <a href="http://en.wikipedia.org/wiki/Topological_sorting">http://en.wikipedia.org/wiki/Topological_sorting</a>
38   */
39  public interface ITopologicalSortStrategy<TVertex extends IVertex> {
40    /**
41     * Given an instance of {@link IAdjacencyList}, topologically sorts the graph.
42     *
43     * @param adjacencyList An instance of {@link IAdjacencyList}.
44     * @return The topologically sorted {@link IGraph}.
45     * @throws CyclicGraphException A {@link CyclicGraphException} is thrown if a cycle is detected during the sort.
46     *
47     * @see <a href="http://en.wikipedia.org/wiki/Topological_sorting">http://en.wikipedia.org/wiki/Topological_sorting</a>
48     */
49    List<TVertex> sort(IAdjacencyList<TVertex> adjacencyList) throws CyclicGraphException;
50  
51    /**
52     * Allows you to asynchronously and in-parallel process the vertices of a graph topologically. Care is taken to
53     * prevent the processing of vertices before their parent vertices have completed processing.
54     *
55     * @param executor      An instance of {@link ExecutorService} that will be used to submit tasks for processing
56     *                      vertices.
57     * @param adjacencyList An instance of {@link IAdjacencyList}.
58     * @param callback      An instance of {@link ITopologicalSortCallback} that will be called to process each vertex.
59     *                      This may be called concurrently depending on the makeup of the graph.
60     * @param errorCallback An instance of {@link ITopologicalSortErrorCallback} that will be called if an error during
61     *                      processing occurs either in the
62     *                      {@link ITopologicalSortCallback#handle(IVertex, ITopologicalSortInput, ITopologicalSortCoordinator)} method or in the
63     *                      {@link ITopologicalSortStrategy#sortAsync(java.util.concurrent.ExecutorService, IAdjacencyList, ITopologicalSortCallback, ITopologicalSortErrorCallback)}
64     *                      method.
65     * @return              An instance of {@link ITopologicalSortAsyncResult} that allows the caller to coordinate the
66     *                      asynchronous processing of the graph.
67     *
68     * @see <a href="http://en.wikipedia.org/wiki/Topological_sorting">http://en.wikipedia.org/wiki/Topological_sorting</a>
69     */
70    ITopologicalSortAsyncResult sortAsync(ExecutorService executor, IAdjacencyList<TVertex> adjacencyList, ITopologicalSortCallback<TVertex> callback, ITopologicalSortErrorCallback<TVertex> errorCallback);
71  }
