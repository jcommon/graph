20  package jcommon.graph;
21  
22  import java.util.ArrayList;
23  import java.util.Arrays;
24  import java.util.List;
25  
26  public class ValidList<TValue extends Object> {
27    private final TValue[] list;
28  
29    public ValidList(TValue... values) {
30      this.list = values;
31    }
32  
33    public TValue[] getList() {
34      return list;
35    }
36  
37    @Override
38    public String toString() {
39      return "" + (list == null ? null : Arrays.asList(list));
40    }
41  
42    public static <TValue extends Object> ValidList build(TValue...values) {
43      return new ValidList<TValue>(values);
44    }
45  
46    public static ValidList<String> buildFromStrings(String...values) {
47      return new ValidList<String>(values);
48    }
49  
50    public static <TValue extends Number> ValidList<TValue> buildFromNumbers(TValue...values) {
51      return new ValidList<TValue>(values);
52    }
53  
54    public boolean matches(TValue...values) {
55      return matches(Arrays.asList(values));
56    }
57  
58    public boolean matches(Iterable<TValue> values) {
59      //Convert to a list first.
60      final List<TValue> l = new ArrayList<TValue>();
61      for (TValue t : values) {
62        l.add(t);
63      }
64      return matches(l);
65    }
66  
67    public boolean matches(List<TValue> values) {
68      if (list.length != values.size())
69        return false;
70  
71      //Order does not matter -- just presence
72  
73      for(TValue v : list) {
74        if (!values.contains(v))
75          return false;
76      }
77  
78      return true;
79    }
80  }
